<script>

let logs = JSON.parse(localStorage.getItem("keiba_pro_logs") || "[]");

function run(){

  let ai1 = +ai1.value;
  let ai2 = +ai2.value;
  let ai3 = +ai3.value;
  let odds = +odds.value;
  let pop = +pop.value;
  let bank = +bank.value;
  let hit = +hit.value;

  if(!ai1 || !ai2 || !ai3 || !odds || !pop){
    result.innerText = "入力不足";
    return;
  }

  let aiDiff = ai1 - ai2;

  // ■ スコア
  let score = 0;

  if(aiDiff >= 7) score += 2;
  else if(aiDiff >= 3) score += 1;

  if(ai1 >= 95) score += 1;

  if(ai2 - ai3 <= 3) score += 1;

  if(odds >= 4) score += 2;
  else if(odds >= 2.5) score += 1;

  if(pop >= 3 && ai1 >= 90) score += 1;

  // ■ グレード
  let grade = "";
  let rate = 0;

  if(score >= 8){
    grade = "S（全力）";
    rate = 0.03;
  }
  else if(score >= 6){
    grade = "A（勝負）";
    rate = 0.02;
  }
  else if(score >= 4.5){
    grade = "B（抑え）";
    rate = 0.01;
  }
  else{
    grade = "C（見送り）";
    rate = 0;
  }

  // ■ 連敗管理
  let loseCount = logs.slice(-5).filter(l => l.hit == 0).length;

  if(loseCount >= 3) rate *= 0.7;
  if(loseCount >= 5) rate *= 0.5;

  // ■ 勝ちモード
  let winStreak = logs.slice(-3).filter(l => l.hit == 1).length;

  if(winStreak >= 3) rate *= 1.2;

  rate = Math.min(rate, 0.05);

  let bet = Math.round(bank * rate);

  let profit = (hit == 1) ? bet * 3 - bet : -bet;

  logs.push({score, grade, bet, profit, hit});

  localStorage.setItem("keiba_pro_logs", JSON.stringify(logs));

  // ■ 集計
  let totalBet = logs.reduce((a,b)=>a+b.bet,0);
  let totalProfit = logs.reduce((a,b)=>a+b.profit,0);
  let roi = totalBet ? ((totalProfit + totalBet)/totalBet*100).toFixed(1) : 0;

  result.innerText =
`判定：${grade}
スコア：${score}
賭け金：${bet}円`;

  money.innerText =
`■ 資金管理
元本：${bank}円
総賭け：${totalBet}円
収支：${totalProfit}円
回収率：${roi}%`;

  model.innerText =
`■ 状態
連敗傾向：${loseCount}
連勝傾向：${winStreak}
リスク自動調整：ON`;

}

</script>
